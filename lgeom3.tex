\section{Definiciones preliminares}
\begin{definicion}{la cuerda de una circunferencia es un segmento que se
interseca con la circunferencia en sus extremos}{Cuerda}

\end{definicion}
\begin{definicion}{El di\'ametro de una circunferencia es una cuerda que
contiene el centro de esta.}{Di\'ametro}
\end{definicion}

\begin{definicion}{Una tangente es la recta que interseca a la circunferencia
en un \'unico punto.}{Recta tangente}

\end{definicion}
\begin{definicion}{Un \'angulo inscrito es un \'angulo con v\'ertice en la
circunferencia y con lados que contienen cuerdas de la circunferencia.}{\'Angulo
inscrito}
\end{definicion}
\begin{definicion}{Una recta secante de una circunferencia es aquella que
interseca a la circunferencia en dos puntos exactamente.}{Recta secante}
\end{definicion}

\begin{definicion}{Un \'angulo en una circunferencia se le llama central si
tiene v\'ertice en el centro, y sus lados contienen a radios de la
circunferencia.}{\'Angulo central}
\end{definicion}
\begin{definicion}{Un arco de una circunferencia se le llama menor si est\'a
en el interior de \'angulo central}{Arco menor}
\end{definicion}
\begin{definicion}{La mediada de un arco menor es la medida del \'angulo
central que lo contiene}{Medida de un arco menor}
\end{definicion}
\begin{definicion}{Un arco de una circunferencia se le llama mayor si es el
complemento del arco menor, con respecto a la circunferencia}{Arco mayor}
\end{definicion}
\begin{definicion}{
Si $\alpha$ es la medida de un arco menor de una circunferencia, la medida
del arco mayor asociado a \'el es: $360^\circ -\alpha$}{Medida del arco mayor}
\end{definicion}
\begin{postulado}{Si Un punto $C$ est\'a en un $\arco[20pt]{AB}$, entonces
$m\arco[20pt]{AC}+m\arco[20pt]{CB}=m\arco[20pt]{AB}$}{Suma de arcos}
 \end{postulado}
\begin{definicion}{Si dos arcos tienen la misma medida, los arcos son
congruentes}{Arcos congruentes}
\end{definicion}
\begin{teorema}{En un circunferencia, o en circunferencias congruentes, las
cuerdas congruentes tiene arcos menores congruentes}{Arcos congruentes}
\end{teorema}
\begin{teorema}{En una circunferencia, o en circunferencias congruentes , los
arcos menores congruentes tienen cuerdas congruentes }{Reciproco}
\end{teorema}
\begin{teorema}{En una circunferencia, o en circunferencias congruentes, las
cuerdas congruentes equidistan del centro}{Cuerdas congruentes}
\end{teorema}

\begin{teorema}{En una circunferencia, o en circunferencias congruentes, las
cuerdas congruentes equidistan del centro}{Cuerdas equidistantes}
\end{teorema}
\begin{teorema}{En una circunferencia, o en circunferencias congruentes, las
cuerdas equidistantes del centro con congruentes}{Reciproco}
\end{teorema}

\begin{teorema}{La bisectriz perpendicular de una cuerda contiene al centro
de la circunferencia}{Bisectriz de una cuerda}
\end{teorema}
\begin{teorema}{Si una recta que pasa por el centro de una circunferencia es
perpendicular a una cuerda que no es un di\'ametro, entonces biseca a la
cuerda y a su arco menor}{}
\end{teorema}

\begin{teorema}{Si una recta recta es tangente a una circunferencia, entonces
 el radio trazado hasta el punto de contacto es perpendicular a la recta
tangente}{}
\end{teorema}

\begin{teorema}{Si una recta es perpendicular a una recta tangente en un
punto de una circunferencia, entonces la recta contiene al centro de la
circunferencia}{}
 \end{teorema}
\begin{teorema}{Los segmentos tangentes a una circunferencia desde un punto
exterior son congruentes y forman \'angulos congruentes con la recta que une
al centro de la circunferencia con el punto exterior}{}
 \end{teorema}
\begin{teorema}{La medida de unde un \'angulo inscrito es la mitad de la
medida de un arco interceptado}{}
\end{teorema}
\begin{teorema}{Un \'angulo inscrito en una semicircunferencia es un \'angulo
recto}{}
\end{teorema}
\begin{teorema}{Un \'angulo formado por dos cuerdas que se intersecan en el
interior de una circunferencia tiene una medida igual a la semisuma de las
medidas de los arcos interceptados}{}
 
\end{teorema}

\begin{teorema}{La medida del \'angulo formato por una recta tangente y una
cuerda trazada al punto de contacto es igual a la mitad de la medida del arco
interceptado}{}
\end{teorema}
\begin{teorema}{La medida de un \'angulo formado por dos rectas tangentes a
